#include <stdio.h>

int main() {
    int numSubjects;
    float totalCreditHours = 0, totalGradePoints = 0, totalSubjectMark = 0, totalObtainMark = 0, cgpa;

    printf("Enter the number of subjects: ");
    scanf("%d", &numSubjects);

    for (int i = 0; i < numSubjects; i++) {
        char subjectName[50];
        int creditHours;
        float subjectMark, obtainMark, percentage;
        float gradePoint;

        printf("\nEnter details for subject %d:\n", i + 1);

        printf("Subject Name: ");
        scanf("%s", subjectName);

        printf("Credit Hours: ");
        scanf("%d", &creditHours);

        printf("Subject Mark: ");
        scanf("%f", &subjectMark);
        totalSubjectMark += subjectMark;

        printf("Obtain Mark: ");
        scanf("%f", &obtainMark);
        totalObtainMark += obtainMark;

        // Calculate percentage
        percentage = (obtainMark / subjectMark) * 100;

        // Calculate grade point based on percentage
        if (percentage >= 80) {
            gradePoint = 4.00;
        } else if (percentage >= 75) {
            gradePoint = 3.75;
        } else if (percentage >= 70) {
            gradePoint = 3.50;
        } else if (percentage >= 65) {
            gradePoint = 3.25;
        } else if (percentage >= 60) {
            gradePoint = 3.00;
        } else if (percentage >= 55) {
            gradePoint = 2.75;
        } else if (percentage >= 50) {
            gradePoint = 2.50;
        } else if (percentage >= 45) {
            gradePoint = 2.25;
        } else if (percentage >= 40) {
            gradePoint = 2.00;
        } else {
            gradePoint = 0.00;
        }

        totalCreditHours += creditHours;
        totalGradePoints += creditHours * gradePoint;
    }

    cgpa = totalGradePoints / totalCreditHours;

    // Determine CGPA Grand Name
    const char *grandName;
    if (cgpa >= 4.00) {
        grandName = "A+";
    } else if (cgpa >= 3.75) {
        grandName = "A";
    } else if (cgpa >= 3.50) {
        grandName = "A-";
    } else if (cgpa >= 3.25) {
        grandName = "B+";
    } else if (cgpa >= 3.00) {
        grandName = "B";
    } else if (cgpa >= 2.75) {
        grandName = "B-";
    } else if (cgpa >= 2.50) {
        grandName = "C+";
    } else if (cgpa >= 2.25) {
        grandName = "C";
    } else if (cgpa >= 2.00) {
        grandName = "D";
    } else {
        grandName = "F";
    }

    printf("\nTotal Subject Mark: %.2f\n", totalSubjectMark);
    printf("Total Obtain Mark: %.2f\n", totalObtainMark);
    printf("Your CGPA is: %.2f\n", cgpa);
    printf("Your CGPA Grand Name is: %s\n", grandName);

    return 0;
}
